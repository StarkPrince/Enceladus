import os

def convert_filenames():
    # Get the current working directory
    cwd = os.getcwd()
    
    # List the first 15 items in the directory
    items = os.listdir(cwd)[:15]
    
    # Check if there are at least 15 items
    if len(items) < 15:
        print(f"Warning: Only {len(items)} items found in {cwd}. Renaming only possible for these items.")
    
    # Rename the items
    for i, item in enumerate(items, start=1):
        if i <= len(items):
            new_name = f"ext_{i:02d}"
            old_path = os.path.join(cwd, item)
            new_path = os.path.join(cwd, new_name)
            try:
                os.rename(old_path, new_path)
                print(f"Renamed '{item}' to '{new_name}.jpg'")
            except OSError as e:
                print(f"Error renaming {item}: {e}")

# Run the function
convert_filenames()
